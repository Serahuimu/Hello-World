四页玫瑰数：一个四位数的每一位的四次方加起来等于这个数本身的数

```c
#include <stdio.h>
#include <math.h>   //包含pow()函数

int main(void)
{
    int i, j;
    for (i = 1000; i < 10000; i++)              //数的范围是所有四位整数
    {
        int sum = 0;
        for (j = 1; j <= i; j *= 10)
        {
            sum += (int)pow(i / j %10, 4);      //得到每一位数，并对其求4次方
        }
        if (sum == i)
            printf("%d\n", i);
    }

    return 0;
}
```
